⚖️ 법적 준수 마이그레이션 실행

개인정보보호법 및 정보통신망법 완전 준수를 위한 데이터베이스 변경사항을 적용합니다.

⚠️ 중요: 반드시 순서대로 실행하세요

각 마이그레이션은 이전 마이그레이션에 의존성이 있습니다. 반드시 007 → 008 순서로 실행해야 오류가 발생하지 않습니다.

1
마케팅 동의 분리 (정보통신망법 제50조)

파일: supabase/migrations/007_marketing_consent_separation.sql

정보통신망법 제50조에 따라 전화(음성), SMS/문자, 이메일 마케팅 수신 동의를 각각 분리하여 수집하도록 테이블 구조를 변경합니다. 

-- 007_marketing_consent_separation.sql

-- profiles 테이블에 마케팅 수신 동의 분리 (정보통신망법 제50조)
ALTER TABLE public.profiles
  DROP COLUMN IF EXISTS marketing_agreement,
  ADD COLUMN IF NOT EXISTS marketing_sms boolean DEFAULT false,
  ADD COLUMN IF NOT EXISTS marketing_email boolean DEFAULT false,
  ADD COLUMN IF NOT EXISTS marketing_phone boolean DEFAULT false,
  ADD COLUMN IF NOT EXISTS marketing_agreed_at timestamptz;

COMMENT ON COLUMN public.profiles.marketing_sms IS '정보통신망법 제50조: SMS/문자 마케팅 수신 동의';
COMMENT ON COLUMN public.profiles.marketing_email IS '정보통신망법 제50조: 이메일 마케팅 수신 동의';
COMMENT ON COLUMN public.profiles.marketing_phone IS '정보통신망법 제50조: 전화(음성) 마케팅 수신 동의';
COMMENT ON COLUMN public.profiles.marketing_agreed_at IS '마케팅 수신 동의 일시';

2
사용자 접근 로그 (개인정보 안전성 확보조치 기준 제7조)

파일: supabase/migrations/008_user_access_logs.sql

개인정보의 안전성 확보조치 기준 제7조에 따라 개인정보 처리 시스템에 접속한 기록을 6개월 이상 보관합니다. 

-- 008_user_access_logs.sql

-- 사용자 접근 로그 테이블 (개인정보의 안전성 확보조치 기준 제7조)
CREATE TABLE IF NOT EXISTS public.user_access_logs (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  action text NOT NULL,
  target_id uuid,
  ip_address inet,
  user_agent text,
  created_at timestamptz DEFAULT now() NOT NULL
);

COMMENT ON TABLE public.user_access_logs IS '사용자 접근 로그 (6개월 보관 의무)';
COMMENT ON COLUMN public.user_access_logs.action IS 'view_profile, update_profile, view_report, create_inquiry, delete_account, view_inquiry';
COMMENT ON COLUMN public.user_access_logs.target_id IS '조회한 리소스 ID (reports.id, inquiries.id 등)';

-- 인덱스: 사용자별 조회 최적화
CREATE INDEX IF NOT EXISTS idx_user_access_logs_user_id ON public.user_access_logs(user_id);
CREATE INDEX IF NOT EXISTS idx_user_access_logs_created_at ON public.user_access_logs(created_at DESC);

-- RLS 활성화
ALTER TABLE public.user_access_logs ENABLE ROW LEVEL SECURITY;

-- 정책: 사용자는 자신의 접근 로그만 조회 가능
CREATE POLICY "Users can view their own access logs"
  ON public.user_access_logs FOR SELECT
  USING (auth.uid() = user_id);

-- 정책: 관리자는 모든 접근 로그 조회 가능
CREATE POLICY "Admins can view all access logs"
  ON public.user_access_logs FOR SELECT
  USING (public.is_admin());

-- 정책: 사용자는 자신의 접근 로그 기록 가능 (INSERT만)
CREATE POLICY "Users can insert their own access logs"
  ON public.user_access_logs FOR INSERT
  WITH CHECK (auth.uid() = user_id);

마이그레이션 검증

모든 마이그레이션이 정상적으로 적용되었는지 확인합니다.

-- 검증 쿼리 1: profiles 테이블에 마케팅 동의 컬럼 존재 확인
SELECT column_name, data_type, is_nullable
FROM information_schema.columns
WHERE table_name = 'profiles'
  AND column_name IN ('marketing_sms', 'marketing_email', 'marketing_phone', 'marketing_agreed_at')
ORDER BY column_name;

-- 예상 결과: 4개 행이 조회되어야 함

-- 검증 쿼리 2: user_access_logs 테이블 존재 확인
SELECT table_name, table_type
FROM information_schema.tables
WHERE table_schema = 'public' AND table_name = 'user_access_logs';

-- 예상 결과: 1개 행 (user_access_logs, BASE TABLE)
✅ 두 쿼리 모두 예상 결과가 나오면 마이그레이션이 성공적으로 완료된 것입니다.
